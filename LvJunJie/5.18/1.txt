import vtk
reader1 = vtk.vtkSTLReader()
reader1.SetFileName("shizhen.stl")
reader2 = vtk.vtkSTLReader()
reader2.SetFileName("fenzhen.stl")
reader3 = vtk.vtkSTLReader()
reader3.SetFileName("yuanpan.stl")
mapper1 = vtk.vtkPolyDataMapper()
mapper1.SetInputConnection(reader1.GetOutputPort())
mapper2 = vtk.vtkPolyDataMapper()
mapper2.SetInputConnection(reader2.GetOutputPort())
mapper3 = vtk.vtkPolyDataMapper()
mapper3.SetInputConnection(reader3.GetOutputPort())
actor1 = vtk.vtkActor()
actor1.SetMapper(mapper1)
actor2 = vtk.vtkActor()
actor2.SetMapper(mapper2)
actor3 = vtk.vtkActor()
actor3.SetMapper(mapper3)
renderer = vtk.vtkRenderer()
renderWindow = vtk.vtkRenderWindow()
renderWindow.AddRenderer(renderer)
renderWindowInteractor = vtk.vtkRenderWindowInteractor()
renderWindowInteractor.SetRenderWindow(renderWindow)
renderer.AddActor(actor1)
renderer.AddActor(actor2)
renderer.AddActor(actor3)
renderer.SetBackground(1,1,1)
actor1.SetOrigin(-50.0, 50.0, 2.0)
print("actor1 center:",actor1.GetCenter())
actor2.SetOrigin(-50.0, 50.0, 2.0)
print("actor2 center:",actor1.GetCenter())
actor3.SetOrigin(-50.0, 50.0, 2.0)
print("actor3 center:",actor1.GetCenter())
actor1.GetProperty().SetColor(1,0,0)
actor2.GetProperty().SetColor(0,1,0)
actor3.GetProperty().SetColor(0,0,1)
for i in range(360000):
    actor1.RotateZ(-0.001)
    actor2.RotateZ(-0.012)
    renderWindow.Render()
